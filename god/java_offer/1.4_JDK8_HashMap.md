## put方法

```java
public V put(K key, V value) {
    return putVal(hash(key), key, value, false, true);
}


final V putVal(int hash, K key, V value, boolean onlyIfAbsent,  boolean evict) {
    Node<K,V>[] tab; Node<K,V> p; int n, i;
    
    // 初始化容量
    if ((tab = table) == null || (n = tab.length) == 0)
        n = (tab = resize()).length;
    
    // 哈希桶中插入第一个元素
    if ((p = tab[i = (n - 1) & hash]) == null)
        tab[i] = newNode(hash, key, value, null);
    else {
        // 哈希桶中有其他元素
        Node<K,V> e; K k;
        if (p.hash == hash && ((k = p.key) == key || (key != null && key.equals(k))))
            e = p; // 数组中的第一个元素等于该key, 修改后返回
        
        else if (p instanceof TreeNode)
        	// 链表升级成红黑树, 使用红黑树放置数据
            // 放置数据的逻辑和下节中的链表树化类似
            e = ((TreeNode<K,V>)p).putTreeVal(this, tab, hash, key, value);
        	
        
        else {
            for (int binCount = 0; ; ++binCount) {
                if ((e = p.next) == null) {
                    // 遍历到链表尾部, 没有发现key, 使用尾插法插入新值
                    p.next = newNode(hash, key, value, null);
                    
                    // 链表上面已经有8(默认阈值)个了, 尝试为红黑树
                    // 不一定转化成红黑树!!!
                    // 因为转化函数会判断MIN_TREEIFY_CAPACITY是否满足, 不满足扩容, 满足才转化为红黑树
                    if (binCount >= TREEIFY_THRESHOLD - 1) // -1 for 1st
                        treeifyBin(tab, hash);
                    
                    break;
                }
                if (e.hash == hash &&
                    ((k = e.key) == key || (key != null && key.equals(k))))
                    break;
                p = e;
            }
        }
        
        // 链表中存在该元素, 替换旧值
        if (e != null) { // existing mapping for key
            V oldValue = e.value;
            if (!onlyIfAbsent || oldValue == null)
                e.value = value;
            afterNodeAccess(e);
            return oldValue;
        }
    }
    
    ++modCount;
    if (++size > threshold)
        resize(); // 扩容
    
    afterNodeInsertion(evict);
    return null;

}
```

## 链表树化

```java
/*
* 把链表全部替换为 TreeNode
*/
final void treeifyBin(Node<K,V>[] tab, int hash) {
    int n, index; Node<K,V> e;
    // 判断当前容量, 是否大于树化的阈值
    if (tab == null || (n = tab.length) < MIN_TREEIFY_CAPACITY)
        resize();
    
    else if ((e = tab[index = (n - 1) & hash]) != null) {
        TreeNode<K,V> hd = null, tl = null;
        do {
            
            TreeNode<K,V> p = replacementTreeNode(e, null);
            if (tl == null)
                hd = p;
            else {
                p.prev = tl;
                tl.next = p;
            }
            tl = p;
           
            
        } while ((e = e.next) != null);
        
        if ((tab[index] = hd) != null)
            hd.treeify(tab);
    
    }
    
}

/**
* 转化链表为红黑树
*/
final void treeify(Node<K,V>[] tab) {
    TreeNode<K,V> root = null;
    for (TreeNode<K,V> x = this, next; x != null; x = next) {
        next = (TreeNode<K,V>)x.next;
        x.left = x.right = null;
        if (root == null) {
            // 把链表的第一位初始化为红黑树的根节点
            x.parent = null;
            x.red = false;
            root = x;
        }
        else {
            
            K k = x.key;
            int h = x.hash;
            Class<?> kc = null; // key的class对象
            
            for (TreeNode<K,V> p = root;;) {
                int dir, ph; // dir 遍历方向(-1 左边, 1右边, 0相等),  ph 节点节点的hash
                K pk = p.key; // 当前节点的key
                if ((ph = p.hash) > h)
                    dir = -1;
                else if (ph < h)
                    dir = 1;
                else if (
                    	(kc == null && (kc = comparableClassFor(k)) == null) 
                         ||
                         (dir = compareComparables(kc, k, pk)) == 0
                        )
                    // hash相同, 调用class的compare对象
                    dir = tieBreakOrder(k, pk);

                TreeNode<K,V> xp = p;
                
                // 决定方向, 如果是叶子节点插入到红黑树中, 否则继续遍历
                if ((p = (dir <= 0) ? p.left : p.right) == null) {
                    x.parent = xp;
                    if (dir <= 0)
                        xp.left = x;
                    else
                        xp.right = x;
                    
                    // 插入红黑树中, 上一节已经解释了
                    root = balanceInsertion(root, x);
                    break;
                }
            }
        }
    }
    moveRootToFront(tab, root); // 把红黑树的头节点放到 哈希数组中
}

static <K,V> void moveRootToFront(Node<K,V>[] tab, TreeNode<K,V> root) {
    int n;
    if (root != null && tab != null && (n = tab.length) > 0) {
        int index = (n - 1) & root.hash;
        TreeNode<K,V> first = (TreeNode<K,V>)tab[index];
        if (root != first) {
            // 数组中存储的元素不是头节点, 替换first节点(双向链表更换节点)
            Node<K,V> rn;
            tab[index] = root;
            TreeNode<K,V> rp = root.prev;
            if ((rn = root.next) != null)
                ((TreeNode<K,V>)rn).prev = rp;
            if (rp != null)
                rp.next = rn;
            if (first != null)
                first.prev = root;
            root.next = first;
            root.prev = null;
        }
        assert checkInvariants(root);
    }
}
```

## 扩容

扩容触发是在存储容量大于阈值

```java
final Node<K,V>[] resize() {
    Node<K,V>[] oldTab = table; // 旧表
    int oldCap = (oldTab == null) ? 0 : oldTab.length; // 旧容量
    int oldThr = threshold; // 旧阈值
    int newCap, newThr = 0; // 新容量, 新阈值
    if (oldCap > 0) {
        // 新容量 左移一位
        if (oldCap >= MAXIMUM_CAPACITY) {
            threshold = Integer.MAX_VALUE;
            return oldTab;
        }
        else if ((newCap = oldCap << 1) < MAXIMUM_CAPACITY &&
                 oldCap >= DEFAULT_INITIAL_CAPACITY)
            newThr = oldThr << 1; // double threshold
    }
    
    else if (oldThr > 0) // initial capacity was placed in threshold
        newCap = oldThr;
    else {               // zero initial threshold signifies using defaults
        newCap = DEFAULT_INITIAL_CAPACITY;
        newThr = (int)(DEFAULT_LOAD_FACTOR * DEFAULT_INITIAL_CAPACITY);
    }
    if (newThr == 0) {
    	// 计算新阈值
        float ft = (float)newCap * loadFactor;
        newThr = (newCap < MAXIMUM_CAPACITY && ft < (float)MAXIMUM_CAPACITY ?
                  (int)ft : Integer.MAX_VALUE);
    }
    // ===== 初始化完成 ==
    
    
    threshold = newThr;
    @SuppressWarnings({"rawtypes","unchecked"})
    Node<K,V>[] newTab = (Node<K,V>[])new Node[newCap];
    table = newTab;
    if (oldTab != null) {
        // 遍历旧数组
        for (int j = 0; j < oldCap; ++j) {
            Node<K,V> e;
            if ((e = oldTab[j]) != null) {
                // 有元素开始转移
                oldTab[j] = null;
                // 当前位置只有一个元素
                if (e.next == null)
                    newTab[e.hash & (newCap - 1)] = e;
                
                else if (e instanceof TreeNode)
                	// 红黑树分解
                    ((TreeNode<K,V>)e).split(this, newTab, j, oldCap);
                
                
                else {
                    
                    // 链表转移
                    // preserve order
                    Node<K,V> loHead = null, loTail = null;
                    Node<K,V> hiHead = null, hiTail = null;
                    Node<K,V> next;
                    
                    // 循环的逻辑是, 把链表分为两组
                    // 一组是 e.hash & oldCap = 0 扩容后的位置不变
                    // 另一组是 e.hash & oldCap = 0 扩容后, 位置 改变
                    // 为什么根据这一个标志位就可以判断, 在于哈希桶算法
                    do {
                        next = e.next;
                        if ((e.hash & oldCap) == 0) {
                            if (loTail == null)
                                loHead = e;
                            else
                                loTail.next = e;
                            loTail = e;
                        }
                        else {
                            if (hiTail == null)
                                hiHead = e;
                            else
                                hiTail.next = e;
                            hiTail = e;
                        }
                    } while ((e = next) != null);
                    
                    
                    // 转移不变的新链表
                    if (loTail != null) {
                        loTail.next = null;
                        newTab[j] = loHead;
                    }
                    
                    // 转移位置改变的链表
                    if (hiTail != null) {
                        hiTail.next = null;
                        newTab[j + oldCap] = hiHead;
                    }
                    
                }
            }
        }
    }
    return newTab;
}


// ==================   红黑树扩容机制 ===============
final void split(HashMap<K,V> map, Node<K,V>[] tab, int index, int bit) {
    TreeNode<K,V> b = this;
    // Relink into lo and hi lists, preserving order
    TreeNode<K,V> loHead = null, loTail = null; // 位置不变的头尾指针
    TreeNode<K,V> hiHead = null, hiTail = null; // 位置改变的头尾指针
    int lc = 0, hc = 0; // 计数
    
    // 把红黑树分为两组, 逻辑同上面的扩容代码
    for (TreeNode<K,V> e = b, next; e != null; e = next) {
        next = (TreeNode<K,V>)e.next;
        e.next = null;
        if ((e.hash & bit) == 0) {
            if ((e.prev = loTail) == null)
                loHead = e;
            else
                loTail.next = e;
            loTail = e;
            ++lc;
        }
        else {
            if ((e.prev = hiTail) == null)
                hiHead = e;
            else
                hiTail.next = e;
            hiTail = e;
            ++hc;
        }
    }

    if (loHead != null) {
        // 位置不变得链表容量小于等于6(默认值), 红黑树退化为链表
        if (lc <= UNTREEIFY_THRESHOLD)
            tab[index] = loHead.untreeify(map);
        else {
            // 容量过多, 重新生成红黑树
            tab[index] = loHead;
            if (hiHead != null) // (else is already treeified)
                loHead.treeify(tab);
        }
    }
    
    // 逻辑同上
    if (hiHead != null) {
        if (hc <= UNTREEIFY_THRESHOLD)
            tab[index + bit] = hiHead.untreeify(map);
        else {
            tab[index + bit] = hiHead;
            if (loHead != null)
                hiHead.treeify(tab);
        }
    }
    
}
```
