## 2. 配置日志
```go
import (
    "flag"
    "mikan/internal/config"
    "mikan/internal/downloader"
    "mikan/internal/message"

    "mikan/internal/resource"
    "mikan/internal/spider"
    "time"

    "io"
    "os"
    "path/filepath"

    log "github.com/sirupsen/logrus"
    "gopkg.in/natefinch/lumberjack.v2"
)

func initLog(logPath string, logLevel string) {
    lumberjackLogger := &lumberjack.Logger{
        Filename:   filepath.ToSlash(logPath),
        MaxSize:    5, // MB
        MaxBackups: 10,
        MaxAge:     30,   // days
        Compress:   true, // disabled by default
    }
    // Fork writing into two outputs
    multiWriter := io.MultiWriter(os.Stdout, lumberjackLogger)
    logFormatter := new(log.TextFormatter)
    logFormatter.TimestampFormat = "2006-01-02 15:04:05" // or RFC3339
    logFormatter.FullTimestamp = true
    log.SetFormatter(logFormatter)
    switch logLevel {
    case "debug":
        log.SetLevel(log.DebugLevel)
    case "info":
        log.SetLevel(log.InfoLevel)
    case "warn":
        log.SetLevel(log.WarnLevel)
    case "error":
        log.SetLevel(log.ErrorLevel)
    }
    log.SetOutput(multiWriter)
}

var (
    configPath string
)

func init() {
    flag.StringVar(&configPath, "c", "./config/prod/config.yml", "set configuration `file`")
}

func main() {
    flag.Parse()
    mikanConfig := config.InitConfig(configPath)
    initLog("/home/work/log/service.log", "debug")
}
```
